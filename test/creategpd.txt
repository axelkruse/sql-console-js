PRAGMA foreign_keys = ON;
CREATE TABLE DETECTIVES
(
  DID Integer NOT NULL,
  NAME Varchar(20) NOT NULL,
  GENDER Char(1),
  TEAMLEADER Integer,
  DRIVERLICENCE CHAR(1),
  PRIMARY KEY (DID),
  FOREIGN KEY( TEAMLEADER) REFERENCES DETECTIVES(DID)
);
CREATE TABLE CARS
(
  LICENCEPLATTE Char(10) NOT NULL,
  CARTYPE Varchar(15),
  DRIVER Integer,
  PRIMARY KEY (LICENCEPLATTE),
  FOREIGN KEY(DRIVER) REFERENCES DETECTIVES(DID)
);
